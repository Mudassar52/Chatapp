Firestore Rules Fix

Real-time chat kaam karne ke liye Firebase Console mein yeh rules set karo:

rules_version = '2'; service cloud.firestore { match /databases/{db}/documents { // Users — anyone logged in can READ all users (for email search) // Only owner can WRITE their own profile match /users/{uid} { allow read: if request.auth != null; allow write: if request.auth.uid == uid; // Contacts — owner can read own contacts // ANY logged-in user can WRITE (needed for cross-user chat creation) match /contacts/{chatId} { allow read: if request.auth.uid == uid; allow write: if request.auth != null && ( request.auth.uid == uid || chatId.matches(request.auth.uid + '_.*') || chatId.matches('.*_' + request.auth.uid) ); } // Statuses match /statuses/{sid} { allow read: if request.auth != null; allow write: if request.auth.uid == uid; } } // Shared chat messages — both participants can read/write match /chats/{chatId} { allow read, write: if request.auth != null && ( chatId.matches(request.auth.uid + '_.*') || chatId.matches('.*_' + request.auth.uid) ); match /messages/{msgId} { allow read, write: if request.auth != null && ( chatId.matches(request.auth.uid + '_.*') || chatId.matches('.*_' + request.auth.uid) ); } // Typing indicators match /meta/{doc} { allow read, write: if request.auth != null && ( chatId.matches(request.auth.uid + '_.*') || chatId.matches('.*_' + request.auth.uid) ); } } } }
